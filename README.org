#+TITLE: JC

Statically compiled programming language.

# Example

#+BEGIN_SRC jc
  distinct type io_result = i32;
  type path = !u8;

  extern "C" fn <void> printf(msg: !u8, ..);
  extern "C" fn <io_result> stat(_: path, _: !statbuf);

  fn <bool> io_result.ok(self) {
      return (self -> i32) == 0;
  }

  fn <i32> main () {
    var stat1: statbuf;

    let err = stat("jcc", &stat1);
    if !err.ok() {
      // Too bad...
      printf("Failed to open file\n");
    } else {
      printf("Yaaayyy, the file exists!\n");
    }
    printf("Return Value: %i\n", (err -> i32));
    return 0;
  }
#+END_SRC
